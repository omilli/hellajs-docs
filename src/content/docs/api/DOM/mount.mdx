---
title: mount
description: Renders a virtual DOM tree to the real DOM.
---

```typescript
/**
 * @param vNode A virtual DOM node or a function that returns a virtual DOM node.
 * @param rootSelector A CSS selector string that specifies the root element, defaults to #app
*/

mount(vNode: VNode | (() => VNode), rootSelector: string)
```

## Example

```typescript
import { mount, html } from '@hellajs/core';
import { signal } from '@hellajs/core';

const { div, h1, button } = html;

const count = signal(0);

const increment = () => {
  count.set(count() + 1);
};

const App = () => {
  return div(
    h1(`Count: ${count()}`),
    button({ onClick: increment }, 'Increment')
  );
};

// Mount the App
mount(App, '#root');
```
