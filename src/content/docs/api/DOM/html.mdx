<p class="subtitle">Create DOM elements using tag functions.</p>

```typescript
/**
 * Factory for DOM elements.
 * @example html.div({ class: "foo" }, "bar")
 */
export const html: {
  [K in keyof HTMLElementTagNameMap]: (
    props?: VNodeProps,
    ...children: VNodeValue[]
  ) => VNode;
} & {
  $: (
    props?: VNodeProps,
    ...children: VNodeValue[]
  ) => VNode;
};
```

## Usage

Use <code>html</code> to create elements, set attributes, and nest children. Supports fragments and arrays.

```typescript
import { html } from '@hellajs/core';
const { div, input, button } = html;

const form = div(
  input({ 
    type: 'text',
    placeholder: 'Enter your name',
    className: 'input-field',
    required: true
  }),
  button({ 
    type: 'submit',
    className: 'btn primary',
    onclick: (e) => {
      console.log('Button clicked!', e);
      e.preventDefault();
    }
  }, 'Submit')
);
```
