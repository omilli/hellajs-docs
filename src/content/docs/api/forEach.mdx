---
title: forEach
description: Loops through an array and renders a template for each item.
---

```typescript
/**
 * @param array - The array to loop through
 * @param arg2 - The template function to render for each item or a key to track the item 
 * @param arg3 - The template function to render for each item if arg2 is a key
 * @returns A function that can be called to stop the loop
 */

type ForEach<T> = (item: T, index: number) => VNodeValue;

export function forEach<T>(
  each: T[] | (() => T[]),
  arg2: ForEach<T> | keyof T,
  arg3?: ForEach<T>
)
```

Loops through an array and renders a template for each item. This is useful for rendering lists of items in a declarative way.

## Example

```typescript
import { html, forEach } from '@hellajs/core';

const { ul, li } = html;

const items = signal([1, 2, 3]);

const List = ul(
  forEach(items, (item) =>
    li({ key: item },
      `Item ${item}`
    )
  )
);
```
