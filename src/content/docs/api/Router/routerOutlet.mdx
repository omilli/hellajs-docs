<p class="subtitle">Reactive outlet for rendering the current route handler.</p>

```typescript
/**
 * Renders the current route's handler reactively.
 * 
 * @returns Node placeholder that updates when the route changes.
 */
routerOutlet(): Node
```

## Usage

Use `routerOutlet()` in your app layout to render the active route handler. The outlet will automatically update when the route changes.

```typescript
import { html, mount } from "@hellajs/core";
import { router, navigate, routerOutlet } from "@hellajs/core";

// Define route handlers
function Home() {
  return html.div(
    html.h2("Home"),
  );
}

function About() {
  return html.div(
    html.h2("About"),
  );
}

// Set up routes
router({
  "/": Home,
  "/about": About,
});

// App layout with routerOutlet
function App() {
  return html.main(
    html.h1("HellaJS Router Example"),
    routerOutlet(),
    html.button({ onclick: () => navigate("/") }, "Home"),
    html.button({ onclick: () => navigate("/about") }, "About")
  );
}

// Mount the app
mount(App, "#app");
```
